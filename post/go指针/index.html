go语法: 指针

  • goland

指针

package main

import "fmt"

type User struct {
	name   string
	age    int
	school string
}

// 返回对象指针
func initUser() *User {
	return &User{
		name: "zhang",
		age:  20,
	}
}
// 类似于成员函数的定义
func (user *User) printInfo() {
	fmt.Printf("%s,%d,%s\n", user.name, user.age, user.school)
}

// 普通函数传参
func printInfo2(user *User) {
	fmt.Printf("%s,%d,%s\n", user.name, user.age, user.school)
}

// 测试函数
func main() {
    // 函数初始化，返回指针
	user := initUser()
	fmt.Printf("%s,%d,%s\n", user.name, user.age, user.school)
	user.printInfo()
	printInfo2(user)

    // 直接初始化
	user2 := User{
		name: "lee",
		age:  28,
	}
	user2.printInfo()
	printInfo2(&user2)
}
